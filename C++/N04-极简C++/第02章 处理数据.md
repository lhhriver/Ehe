# 基本数据类型

## 【例2-1】检测基本数据类型所占的字节数

```c++
//文件:ex2_1.cpp
#include <iostream>
using namespace std;
int main()
{
    cout << "short:  " << sizeof(short) << endl;
    cout << "int:  " << sizeof(int) << endl;
    cout << "long:  " << sizeof(long) << endl;
    cout << "long long:  " << sizeof(long long) << endl;
    cout << "float:  " << sizeof(float) << endl;
    cout << "double:  " << sizeof(double) << endl;
    cout << "long double:  " << sizeof(long double) << endl;
    cout << "char:  " << sizeof(char) << endl;
    cout << "bool:  " << sizeof(bool) << endl;
    return 0;
}

```

输出：

```
short:  2
int:  4
long:  4
long long:  8
float:  4
double:  8
long double:  16
char:  1
bool:  1
```

## 常量

### 【例2-2】整型常量的表示

```c++
//文件:ex2_2.cpp
#include <iostream>
using namespace std;
int main()
{
    cout << 100 << endl; //十进制表示的整型常量
    cout << -200 << endl;
    cout << 034 << endl;   //八进制表示的整型常量  28
    cout << 076 << endl;   // 62
    cout << -023 << endl;  //-19
    cout << 0X3B << endl;  //十六进制表示的整型常量  59
    cout << 0x3F << endl;  // 63
    cout << -0x23 << endl; //-35
    return 0;
}
```

输出：

```
100
-200
28
62
-19
59
63
-35
```

### 【例2-3】带后缀的整型常量

```c++
//文件:ex2_3.cpp
#include <iostream>
using namespace std;
int main()
{
    cout << 100 << endl;            // int
    cout << 100U << endl;           // unsigned
    cout << 100L << endl;           // long
    cout << 100LL << endl;          // long long
    cout << sizeof(100) << endl;    // 100占用的字节数
    cout << sizeof(100U) << endl;   // 100U占用的字节数
    cout << sizeof(100L) << endl;   // 100L占用的字节数
    cout << sizeof(100UL) << endl;  // 100UL占用的字节数
    cout << sizeof(100LL) << endl;  // 100LL占用的字节数
    cout << sizeof(100ULL) << endl; // 100ULL占用的字节数
    return 0;
}
```

输出：

```
100
100
100
100
4
4
4
4
8
8
```

### 【例2-4】实型常量的表示

```c++
//文件:ex2_4.cpp
#include <iostream>
using namespace std;
int main()
{
    cout << 2.18 << endl;      //小数形式
    cout << -5.6 << endl;      //小数形式
    cout << 2.15E+4 << endl;   //指数形式
    cout << 2.15E+20 << endl;  //指数形式
    cout << -23.4E-12 << endl; //指数形式
    return 0;
}
```

输出：

```
2.18
-5.6
21500
2.15e+020
-2.34e-011
```
