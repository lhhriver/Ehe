    cout << "long double:  " << sizeof(long double) << endl;
    cout << "char:  " << sizeof(char) << endl;
    cout << "bool:  " << sizeof(bool) << endl;
    return 0;
}

```

```

int:  4
long:  4
long long:  8
float:  4
double:  8
long double:  16
char:  1
bool:  1

### 变量

#### **【例2-10】变量的初始化与使用**

```c++
//文件:ex2_10.cpp
#include <iostream>
using namespace std;
int main()
{
    float a = 1.2, b(2.3), c;                    //变量a、b初始化、c未初始化
    int d = 10;                                  //变量d初始化为10
    int e;                                       //变量e未初始化
    
    cout << a << " " << b << " " << d << endl;   //输出变量的值
    cout << c << " " << e << endl;               // 输出未初始化的变量
    
    a = a + b;                                   // 变量a加b的结果再赋给变量a，a变成3.5 （1.2+2.3）
    c = a + 20.5;                                // 变量a加20.5的结果赋给变量c，c的值是24（3.5+20.5）
    e = d + 100;                                 // 变量d加100的结果赋给变量e，e的值是110（10+100）
    
    cout << a << "  " << b << "  " << c << endl; // 再输出a、b、c的值
    cout << d << "  " << e << endl;              // 输出d、e得值
    return 0;
}
```

输出：

```
1.2 2.3 10
1.67613e-037 0
3.5  2.3  24
10  110
```

#### **【例2-12】字符参与算术运算**

```c++
//文件:ex2_12.cpp
#include <iostream>
using namespace std;
int main()
{
    char c1 = 'A', c2 = 'a', c3, c4, c5;
    c3 = c1 + 1;  // 'A'的ASCII码加1，是'B'的ASCII码
    c4 = c1 + 10; // 'A'的ASCII码加10，是'K'的ASCII码
    c5 = c4 - 5;  // 'K'的ASCII码减5，是'F'的ASCII码
    
    cout << c1 << " " << c2 << " " << c3 << " ";
    cout << c4 << " " << c5 << endl;

    c3 = c2 + 1;  // 'a'的ASCII码加1，是'b'的ASCII码
    c4 = c2 + 10; // 'a'的ASCII码加10，是'k'的ASCII码
    c5 = c4 - 5;  // 'k'的ASCII码减5，是'f'的ASCII码
    
    cout << c1 << " " << c2 << " " << c3 << " ";
    cout << c4 << " " << c5 << endl;
    cout << c2 - c1 << endl; //两个字符的ASCII相减
    
    char c6 = c1 - 25;       //'A'的ASCII码减25，是'('的ASCII码
    cout << c6 << endl;
    return 0;
}
```

输出：

```
A a B K F
A a b k f
32
(
```



#### **【例2-13】自増、自减运算符的使用**

### 关系运算符

#### **【例2-17 】求关系表达式的值**

```c++
//文件:ex2_17.cpp
#include <iostream>
using namespace std;
int main()
{
    int a, b;
    a = 4;
    b = 10;
    
    cout << "a>b ? " << (a > b) << endl;  // false
    cout << "a>=b? " << (a >= b) << endl; // false
    cout << "a<b ? " << (a < b) << endl;  // true
    cout << "a<=b? " << (a <= b) << endl; // true
    cout << "a==b? " << (a == b) << endl; // false
    cout << "a!=b? " << (a != b) << endl; // true
    return 0;
}
```

输出：

```
a>b ? 0
a>=b? 0
a<b ? 1
a<=b? 1
a==b? 0
a!=b? 1
```



### 逻辑运算符

#### **【例2-18 】逻辑运算符应用实例**

```c++
//нд╪Ч:ex2_18.cpp
#include <iostream>
using namespace std;
int main()
{
    int a = 4, b = 10, c = 20;
    cout << "(a>b)&&(c>b)?   " << ((a > b) && (c > b)) << endl;   // false
    cout << "(a>b)||(c>b)?   " << ((a > b) || (c > b)) << endl;   // true
    cout << "(a!=b)&&(b!=c)? " << ((a != b) && (b != c)) << endl; // true
    cout << "!(c>b)?         " << !(c > b) << endl;               // false
    cout << "!(a>b)||(c>b)?  " << (!(a > b) || (c > b)) << endl;  // true
    cout << "(a>0)&&(c>b)?   " << ((a > 0) && (c > b)) << endl;   // true
    return 0;
}
```

输出：

```
(a>b)&&(c>b)?   0
(a>b)||(c>b)?   1
(a!=b)&&(b!=c)? 1
!(c>b)?         0
!(a>b)||(c>b)?  1
(a>0)&&(c>b)?   1
```

### **【例2-20】输出一个三位数的个位数、十位数和百位数**

```c++
//文件:ex2_20.cpp
#include <iostream>
using namespace std;
int main()
{
    int number, units, tens, hundreds; //三位数、个位、十位、百位
    cout << "请输入一个三位数：";
    cin >> number;
    
    hundreds = number / 100; //整数相除结果为整数，小数被舍掉
    tens = (number - 100 * hundreds) / 10;
    units = number - 100 * hundreds - 10 * tens;
    
    cout << "这个数的百位、十位、个位分别是：";
    cout << hundreds << ", " << tens << ", ";
    cout << units << endl;
    return 0;
}
```

输出：

```
请输入一个三位数：258
这个数的百位、十位、个位分别
是：2, 5, 8
```
