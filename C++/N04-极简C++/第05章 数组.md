#### **【例5-5】冒泡排序法**

```c++
//文件:ex5_5.cpp
#include <iostream>
#include <iomanip> //使用setw要包含的头文件
using namespace std;
int main()
{
    int i, j, t, a[10];
    cout << "请输入10个整数，用空格分隔：";
    
    for (i = 0; i < 10; i++)
        cin >> a[i];
    
    for (i = 0; i < 10 - 1; i++) // 10个数需要循环9次
    {
        for (j = 0; j < 10 - 1 - i; j++) //对相邻的两个数进行比较和交换
        {
            if (a[j] > a[j + 1]) //如果前面的数大于后面的数
            {
                t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }
    
    cout << "排序后的数据：";
    for (i = 0; i < 10; i++)
        cout << setw(5) << a[i]; //输出宽度设置为5
    cout << endl;
    
    return 0;
}

```

输出：

```
请输入10个整数，用空格分隔：45 12 65 98 46 52 12 58 63 95
排序后的数据：   12   12   45   46   52   58   63   65   95   98
```

    int a[4][4] = {{11, 24, 53, 14},
                   {51, 36, 27, 18},
                   {29, 15, 41, 62},
                   {23, 84, 75, 26}};
    int i, j, max;
    
    for (i = 0; i < 4; i++) //输出二维数组所有元素的值
    {
        for (j = 0; j < 4; j++)
            cout << setw(4) << a[i][j];

#include <iostream>
#include <cstring>
using namespace std;
int main()
{
    char s[80];
    cout << "请输入一个字符串：" << endl;
    cin >> s;
    int i, n, flag = 1; // flag是1表示是回文，是0表示不是回文
    n = strlen(s);
    
    for (i = 0; i < n / 2; i++)
    {
        if (s[i] != s[n - 1 - i]) //如果 s[i]与s[n-1-i]不相同，不是回文
        {
            flag = 0;
            break;
        }
    }
    if (flag == 1)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;
    return 0;
}

### **【例5-22】折半查找法（二分查找法）**

```c++
//文件:ex5_22.cpp
#include <iostream>
using namespace std;

int main()
{
    int a[10] = {2, 3, 7, 12, 16, 35, 67, 68, 90, 98};
    int mid, low, high, num, index;
    cout << "请输入要查找的数：";
    cin >> num;
    low = 0;       //搜索范围的最小下标
    high = 10 - 1; //搜索范围的最大下标
    index = -1;
    
    while (low <= high) // low<=high表示整个数组尚未查找完
    {
        mid = (low + high) / 2; //求中间元素的下标
        if (num == a[mid])
        {
            index = mid; //若找到,记录下标，结束循环
            break;
        }
        else if (num < a[mid])
            high = mid - 1; //若num<a[mid]将查找范围缩小到数组的前一半
        else
            low = mid + 1; //否则将查找范围缩小到数组的后一半
    }
    if (index == -1) //没有找到
        cout << "未找到\n";
    else
        cout << "找到，下标是：" << index << endl;
    return 0;
}

```

输出：

```
请输入要查找的数：35
找到，下标是：5
```
### **【例5-23】将字符数组中的字符分类**

```c++
//文件:ex5_23.cpp
#include <iostream>
using namespace std;
int main()
{
    char a[20] = "1a3,HA5d7?9mtBr13";
    char b[20]; //保存数字
    char c[20]; //保存字母
    char d[20]; //保存其他字符
    int i, j = 0, k = 0, m = 0;
    
    for (i = 0; a[i] != '\0'; i++) //对数组a循环
    {
        if (isdigit(a[i])) //函数isdigit()判断参数数否为数组字符，是则返回true
        {
            b[j] = a[i];
            j++;
        }
        else if (isalpha(a[i]))
        {
            c[k] = a[i];
            k++;
        }
        else
        {
            d[m] = a[i];
            m++;
        }
    }
    
    b[j] = '\0';
    c[k] = '\0';
    d[m] = '\0';
    
    cout << a << endl;
    cout << b << endl;
    cout << c << endl;
    cout << d << endl;
    return 0;
}

```

输出：

```
1a3,HA5d7?9mtBr13
1357913
aHAdmtBr
,?
```
