### if语句的嵌套

#### **【例3-5】输入任意三个整数a、b、c，输出其中最大的数**

```c++
//文件:ex3_5.cpp
#include <iostream>
using namespace std;
int main()
{
    int a, b, c, max;
    cout << "请输入3个整数，以空格分隔：";
    cin >> a >> b >> c;

    if (a > b) //如果a大于b，则最大值在a和c中找
    {
        if (a > c)
            max = a;
        else
            max = c;
    }
    else //否则，则最大值在b和c中找
    {
        if (b > c)
            max = b;
        else
            max = c;
    }
    
    cout << "max = " << max << endl;
    return 0;
}
```

输出：

```
请输入3个整数，以空格分隔：26 58 96
max = 96
```

### 综合实例

#### **【例3-7】计算指定月份的天数**

```c++
//文件:ex3_7.cpp
#include <iostream>
using namespace std;
int main()
{
    int year, month;
    int days;
    cout << "请输入年份和月份,以空格分隔年份和月份：";
    cin >> year >> month;
    // 1,3,5,7;8,10,12月
    if ((month <= 7 && month % 2 != 0) || (month >= 8 && month % 2 == 0)) // 1,3,5,7 ,8,10,12     31
        days = 31;
    else if (month == 2) // 2月
    {
        if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) //是闰年
            days = 29;
        else //不是闰年
            days = 28;
    }
    else // 4,6,9,11月
        days = 30;

    cout << year << "年 " << month << "月有 " << days << "天" << endl;
    return 0;
}

```

输出：

```
输入年份和月份,以空格分隔年份和月份：2020 10
2020年 10月有 31天
```



#### **【例3-8】求一元二次方程的根**

```c++
//文件:ex3_8.cpp
#include <iostream>
#include <cmath> //数学函数所在的头文件
using namespace std;

int main()
{
    double a, b, c, disc, x1, x2, p, q;
    cout << "请输入三个系数a，b，c，空格分隔：";
    cin >> a >> b >> c;

    if (fabs(a) < 1e-6) //表示a==0,由于实数有误差，不能写成 if(a==0)
        cout << "不是一元二次方程" << endl;
    else // a不等于0
    {
        disc = b * b - 4 * a * c;

        if (fabs(disc) <= 1e-6) //有两个相等的实根
            cout << "x1=x2=" << -b / (2 * a) << endl;
        else
        {
            if (disc > 1e-6) // 有两个不相等的实根
            {
                x1 = (-b + sqrt(disc)) / (2 * a);
                x2 = (-b - sqrt(disc)) / (2 * a);
                cout << "x1= " << x1 << endl;
                cout << "x2= " << x2 << endl;
            }
            else //  有两个共轭复根
            {
                p = -b / (2 * a);
                q = sqrt(fabs(disc)) / (2 * a);
                cout << "x1= " << p << "+" << q << "i" << endl;
                cout << "x2= " << p << "-" << q << "i" << endl;
            }
        }
    }
    return 0;
}

```

输出：

```
请输入三个系数a，b，c，空格分隔：26 86 96
x1= -1.65385+0.978315i
x2= -1.65385-0.978315i
```

#### **【例3-9】用switch语句输出成绩的等级**

```c++
//文件:ex3_9.cpp
#include <iostream>
using namespace std;
int main()
{
    int s, i;
    cout << "请输入成绩：";
    cin >> s;
    
    i = s / 10; // 如果s>=90,i=9或10; 如果s>=80且小于90，i=8;....    1,0
    
    switch (i)
    {
    case 10: // i=10,从这行开始执行
    case 9:
        cout << "优 " << endl; // i=9,从这行开始执行
        break;
    case 8:
        cout << "良 " << endl; // i=8,从这行开始执行
        break;
    case 7:
        cout << "中 " << endl; // i=7,从这行开始执行
        break;
    case 6:
        cout << "及格 " << endl; // i=6,从这行开始执行
        break;
    default:
        cout << "不及格 " << endl; // i=其他值,从这行开始执行
        break;
    }
    return 0;
}

```

```
请输入成绩：56
不及格
```

#### **【例3-10】根据收入计算应纳税款**

```c++
//文件:ex3_10.cpp 
#include <iostream>
#include <cmath>	//数学函数所在的头文件 
using namespace std;

int main()
{
	int salary, i;
	double tax; 
	cout << "输入月收入：";
	cin >> salary;
    
	if(salary < 0)
	{
		cout << "输入有误：\n";
		return 0;
	}
	//salary<5000,  [5000,10000), [10000,20000),[20000,30000)
	//      i=0,     1,             2,3           4,5
    
	i=salary/5000;	
    
	switch(i)
	{
	case 0:		//收入小于5000 
		tax = 0;
		break;
	case 1:		//收入[5000,10000) 
		tax = (salary-5000)*0.05;
		break;
	case 2:		//收入[10000,15000) 
	case 3:		//收入[15000,20000) 
		tax = 5000 * 0.05 + (salary-10000)*0.1;
		break;
	case 4:		//收入[20000,25000) 
	case 5:		//收入[25000,30000) 
		tax = 5000 * 0.05 + 10000*0.1 + (salary-20000)*0.15;
		break;
	default:	//收入大于或等于30000		
		tax = 5000 * 0.05 + 10000*0.1 + 10000*0.15 + (salary-30000)*0.2;
		break;
	}
	cout << "月收入 " << salary << "应缴税额：" << tax << endl;
	return 0; 
}
```

输出：

```
输入月收入：20862
月收入 20862应缴税额：1379.3
```
