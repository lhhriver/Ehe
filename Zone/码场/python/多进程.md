from multiprocessing import Process, cpu_count

MAX_WORKERS_NUM = cpu_count()

li = ["好好学习，天天向上"] * 100000000


def say():

        sum = 0
        for i in range(10000000):

if __name__ == '__main__':
    for i in range(MAX_WORKERS_NUM):
